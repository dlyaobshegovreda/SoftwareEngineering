```python
text = input("текст: ")
print(f"длина: {len(text)}")
lower_text = text.lower()
print(f"в ниж рег: {lower_text}")

vow = "aeiou"
for vow in lower_text:
    if vow in vow:
        count += 1

print("'The': ", text.startswith("The"))
```
